        var absorbRate = _.get(lh, '当前吸收效率', '无近期注入');

        function setBar(id, value, dangerThreshold, invertDanger) {
            var pct = Math.min(100, Math.max(0, value));
            $('#' + id + '-fill').css('width', pct + '%');
            $('#' + id + '-val').text(value);
            $('#' + id + '-fill').removeClass('danger');
            if (invertDanger) {
                if (value <= dangerThreshold) $('#' + id + '-fill').addClass('danger');
            } else {
                if (value >= dangerThreshold) $('#' + id + '-fill').addClass('danger');
            }
        }

        setBar('lh-bar-sanity', sanity, 30, true);
        setBar('lh-bar-heat', heat, 60, false);
        setBar('lh-bar-lewd', lewd, 999, false);
        setBar('lh-bar-orgasm', orgasm, 999, false);
        setBar('lh-bar-energy', energy, 30, true);
        setBar('lh-bar-fluid', fluid, 60, false);

        $('#lh-tag-mental').attr('class', 'lh-tag ' + mentalClass).html('<span class="lh-tag-dot" style="background:' + mentalDotColor + '; box-shadow:0 0 4px ' + mentalDotColor + ';"></span>心理限制：' + mentalWall);

        var heatClass = 'tag-heat';

        else if (heat < 40) { heatStageName = '体温微升'; }
        else if (heat < 60) { heatStageName = '肉体叛离'; heatClass += ' heat-warn'; }
        else if (heat < 80) { heatStageName = '强制发情'; heatClass += ' heat-danger'; }
        else { heatStageName = '失控淫热'; heatClass += ' heat-danger'; }

    <script type="module">
    async function init() {
        await waitGlobalInitialized('Mvu');

        var all_variables = getAllVariables();
        var intelData = _.get(all_variables, 'stat_data.情报库', {});
        var entries = _.entries(intelData);

        $('#intel-count').text(entries.length + ' 条');

        if (entries.length === 0) {
            $('#intel-body').html('<div class="intel-empty">暂无情报记录 — 等待外勤数据回传</div>');
            $('#intel-filters').hide();
            $('#intel-footer').hide();
        } else {
            var classCounts = {};
            var threatCounts = {};
            entries.forEach(function([id, r]) {
                classCounts[r.分类] = (classCounts[r.分类] || 0) + 1;
                threatCounts[r.威胁] = (threatCounts[r.威胁] || 0) + 1;
            });

            var filterHtml = '<button class="filter-btn active" data-filter="all">全部</button>';
            ['植物型', '动物型', '类人型', '微型侵入体', '嵌合体', '未分类'].forEach(function(cls) {
                if (classCounts[cls]) {
                    filterHtml += '<button class="filter-btn" data-filter="' + cls + '">' + cls + ' (' + classCounts[cls] + ')</button>';
                }
            });
            $('#intel-filters').html(filterHtml);

            function renderEntries(filter) {
                var html = '';
                entries.forEach(function([id, r]) {
                    if (filter !== 'all' && r.分类 !== filter) return;
                    html += '<div class="intel-entry" data-class="' + r.分类 + '">';
                    html += '<div class="intel-row-main">';
                    html += '<span class="intel-id">' + id + '</span>';
                    html += '<span class="intel-name">' + r.名称 + '</span>';
                    html += '<div class="intel-tags">';
                    html += '<span class="tag tag-class">' + r.分类 + '</span>';
                    html += '<span class="tag tag-threat-' + r.威胁 + '">' + r.威胁 + '</span>';
                    html += '</div></div>';
                    html += '<div class="intel-row-detail">';
                    html += '<span class="intel-meta">' + (r.记录时间 || '?') + ' · ' + (r.记录地点 || '?') + '</span>';
                    html += '<span class="intel-desc">' + (r.简述 || '—') + '</span>';
                    html += '</div></div>';
                });
                if (!html) {
                    html = '<div class="intel-empty">该分类下暂无记录</div>';
                }
                $('#intel-body').html(html);
            }
            renderEntries('all');

            $(document).on('click', '.filter-btn', function() {
                $('.filter-btn').removeClass('active');
                $(this).addClass('active');
                renderEntries($(this).data('filter'));
            });

            var footerHtml = '';
            ['低级', '中级', '高级', '领主级'].forEach(function(t) {
                if (threatCounts[t]) {
                    footerHtml += t + ':<span>' + threatCounts[t] + '</span>  ';
                }
            });
            $('#intel-stats').html(footerHtml || '暂无威胁分级数据');
        }

        $('#intel-content').hide();
        $('#intel-toggle-icon').addClass('collapsed');

        $('#intel-header').on('click', function() {
            $('#intel-content').slideToggle(200);
            $('#intel-toggle-icon').toggleClass('collapsed');
        });
    }

    $(errorCatched(init));
    </script>
